.PROC,CYBIS. 
.* 
.*  PRIME TIME LOAD PROCEDURE. 
.* 
.*        IF FILE *LCONFIG/PLATOMF* DOES NOT EXIST,
.*                                             LOAD MASTOR.
.*        IF FILE *LCONFIG/PLATOMF* EXISTS, LOAD MASTORN.
.* 
IFE(OT.NE.SYO,ORIGIN)
REVERT,ABORT.
ENDIF(ORIGIN)
.* 
.*        SYSEDIT IN CHANGES IF FILE *SYSEDT* EXISTS.
.* 
SETPUN.
GET(SYSEDT/NA) 
IFE(FILE(SYSEDT,AS),SYSEDIT) 
BEGIN(SYSEDT,SYSEDT) 
UNLOAD(*)
ENDIF(SYSEDIT) 
.* 
.*        SET UP FILE *CONFIGX* UNDER USER *PLATOMF*.
.*        IN THIS PROC, JUST PURGE *CONFIGX*.
.* 
.*SETPUN.
.*PURGE(CONFIGX/NA)
.* 
.*PACKNAM. 
.*CLEAR. 
.* 
SETPUN.
GET,LCONFIG/NA.
CLASS(SC=SY) 
NORERUN. 
SETTL(*) 
SETASL(*)
SETJSL(*)
SET(R1=0)
IFE(FILE(LCONFIG,EQ).EQ.0,MASTOR)
UNLOAD(*)
CONFIGX(V=CYBIS) 
MFNX.
MASTOR(CYBIS)
ELSE(MASTOR) 
UNLOAD(*)
CONFIGX(V=CYBIS) 
MASTORN. 
ENDIF(MASTOR)
DAYFILE. 
EXIT.
DMP. 
CMDMP. 
NOEXIT.
REWIND(OUTPUT) 
PACK(OUTPUT,XP)
RETURN(OUTPUT) 
*
SETPUN.
PACKNAM. 
CHANGE(MASTCM/CE,NA) 
ATTACH(CMDUMP=MASTCM/M=W,NA) 
IFE(FILE(CMDUMP,AS),EXIST) 
UNLOAD(CMDUMP) 
*  MASTOR DUMP FILE ALREADY EXISTS.
$WAIT.  GO TO OVERWRITE DUMP.
PURGE(MASTCM/NA) 
ENDIF(EXIST) 
DEFINE(CMDUMP=MASTCM/M=W)
EMDTAPE.
